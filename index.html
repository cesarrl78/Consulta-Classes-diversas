Exemplos de Classes em Diferentes Linguagens

Exemplo em C++

#include <iostream>
using namespace std;

class MinhaClasseCPlusPlus {
public:
    void Metodo() {
        cout << "Olá do C++!" << endl;
    }
};

int main() {
    MinhaClasseCPlusPlus objeto;
    objeto.Metodo();
    return 0;
}

Exemplo em C

#include <stdio.h>

struct MinhaClasseC {
    int atributo;
};

int main() {
    struct MinhaClasseC objeto;
    objeto.atributo = 42;
    printf("Valor do atributo: %d\n", objeto.atributo);
    return 0;
}

Exemplo em Delphi

unit MinhaClasseDelphi;

interface

type
  TMinhaClasseDelphi = class
  public
    procedure Metodo;
  end;

implementation

procedure TMinhaClasseDelphi.Metodo;
begin
  WriteLn('Olá do Delphi!');
end;

end.
